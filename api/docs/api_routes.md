# API routes

base url: `domain.com/api/`

# GET /feed

- feed data
- get recent posts from following
- paginate

# GET /:address

- profile page
- posts, user profile info (bio, pfp, following, followers)

# GET /post/:id

- post info
- nft, description
- author
- likes
- comments

# POST /user

- info about user
- bio username pfp

# POST /post/:id/comment

- comment

# /post/:id/like

- like

# /user/login

- metamask login
